'use client';

import { useState, useEffect } from 'react';

export default function ThemeToggle() {
  const [theme, setTheme] = useState<'latte' | 'mocha'>('latte');

  useEffect(() => {
    // Read from DOM — the inline script in layout.tsx already set data-theme before paint
    const current = document.documentElement.getAttribute('data-theme');
    setTheme(current === 'mocha' ? 'mocha' : 'latte');
  }, []);

  const toggle = () => {
    const next = theme === 'latte' ? 'mocha' : 'latte';
    setTheme(next);
    document.documentElement.setAttribute('data-theme', next === 'mocha' ? 'mocha' : '');
    localStorage.setItem('theme', next);
  };

  return (
    <button
      onClick={toggle}
      className="px-3 py-1.5 rounded-lg bg-surface0 text-text hover:bg-surface1 transition-colors text-sm font-medium"
      aria-label={`Switch to ${theme === 'latte' ? 'dark' : 'light'} theme`}
    >
      {theme === 'latte' ? 'Mocha' : 'Latte'}
    </button>
  );
}
```

import fs from "fs";
import path from "path";
import { parse } from "csv-parse/sync";
import { SchoolSchema, type School } from "./schema";

let cachedSchools: School[] | null = null;

export function loadSchools(): School[] {
  if (cachedSchools) return cachedSchools;

  const csvPath = path.join(process.cwd(), "data", "usnews-schools.csv");
  const fileContent = fs.readFileSync(csvPath, "utf-8");
  const records = parse(fileContent, {
    columns: true,
    skip_empty_lines: true,
  });

  const schools = records.map((record: any) => ({
    name: record["institution.displayName"],
    state: record["institution.state"],
    city: record["institution.city"],
    ranking: parseInt(record["ranking.displayRank"]) || 999,
    tuitionInState: parseFloat(record["searchData.tuitionInState.rawValue"]) || 0,
    tuitionOutOfState: parseFloat(record["searchData.tuitionOutOfState.rawValue"]) || 0,
    acceptanceRate: parseFloat(record["searchData.acceptanceRate.rawValue"]) || 0,
    percentReceivingAid: parseFloat(record["searchData.percentReceivingAid.rawValue"]) || 0,
    enrollment: parseInt(record["searchData.enrollment.rawValue"]) || 0,
    website: record["School Website"] || "https://example.com",
  }));

  cachedSchools = schools.map((school) => SchoolSchema.parse(school));
  return cachedSchools;
}

export function filterSchoolsByState(schools: School[], state: string): School[] {
  return schools.filter((school) => school.state.toLowerCase() === state.toLowerCase());
}

export function getTopSchools(schools: School[], count: number = 100): School[] {
  return [...schools].sort((a, b) => a.ranking - b.ranking).slice(0, count);
}
```

import fs from "fs";
import path from "path";
import { EarningsSchema, BLSSalarySchema, type Earnings, type BLSSalary } from "./schema";

let cachedEarnings: Earnings[] | null = null;
let cachedBLS: BLSSalary[] | null = null;

export function loadEarnings(): Earnings[] {
  if (cachedEarnings) return cachedEarnings;

  const jsonPath = path.join(process.cwd(), "data", "cs-earnings.json");
  const content = fs.readFileSync(jsonPath, "utf-8");
  const data = JSON.parse(content);

  cachedEarnings = data.map((item: any) => EarningsSchema.parse(item));
  return cachedEarnings;
}

export function loadBLSSalaries(): BLSSalary[] {
  if (cachedBLS) return cachedBLS;

  const jsonPath = path.join(process.cwd(), "data", "bls-salaries.json");
  const content = fs.readFileSync(jsonPath, "utf-8");
  const data = JSON.parse(content);

  cachedBLS = data.map((item: any) => BLSSalarySchema.parse(item));
  return cachedBLS;
}

export function findEarningsForSchool(schoolName: string): Earnings | undefined {
  const earnings = loadEarnings();
  return earnings.find((e) => e.schoolName.toLowerCase().includes(schoolName.toLowerCase()));
}
```

const FEES_PER_YEAR = 2000;
const LIVING_EXPENSES_PER_YEAR = 20000;

export function calculateROI(school: School, options: ROIOptions): ROIResult {
  const { major, inState, commute, startingSalary, interestRate, loanTerm } = options;

  const tuitionPerYear = inState ? school.tuitionInState : school.tuitionOutOfState;
  const housingPerYear = commute ? 0 : HOUSING_COST_PER_YEAR;
  const feesPerYear = FEES_PER_YEAR;

  // Scale aid estimate by the school's percentReceivingAid
  const estimatedFinancialAid = tuitionPerYear * school.percentReceivingAid * 0.4;

  const annualCost = tuitionPerYear + housingPerYear + feesPerYear - estimatedFinancialAid;
  const net4YearCost = annualCost * 4;

  // Amortization formula for monthly loan payment
  const monthlyRate = interestRate / 100 / 12;
  const numberOfPayments = loanTerm * 12;
  const totalLoanAmount = net4YearCost;
  const monthlyLoanPayment =
    totalLoanAmount > 0 && monthlyRate > 0
      ? (totalLoanAmount * monthlyRate * Math.pow(1 + monthlyRate, numberOfPayments)) /
        (Math.pow(1 + monthlyRate, numberOfPayments) - 1)
      : totalLoanAmount > 0
        ? totalLoanAmount / numberOfPayments
        : 0;

  const salaryToDebtRatio = totalLoanAmount > 0 ? startingSalary / totalLoanAmount : Infinity;

  const annualNetEarnings = startingSalary - LIVING_EXPENSES_PER_YEAR;
  const yearsToBreakEven = annualNetEarnings > 0 ? totalLoanAmount / annualNetEarnings : Infinity;

  return {
    schoolName: school.name,
    major,
    tuitionPerYear,
    housingPerYear,
    feesPerYear,
    estimatedFinancialAid: Math.round(estimatedFinancialAid),
    net4YearCost: Math.round(net4YearCost),
    monthlyLoanPayment: Math.round(monthlyLoanPayment),
    totalLoanAmount: Math.round(totalLoanAmount),
    startingSalary,
    salaryToDebtRatio: Math.round(salaryToDebtRatio * 10) / 10,
    yearsToBreakEven: Math.round(yearsToBreakEven * 10) / 10,
    inState,
    commute,
    interestRate,
    loanTerm,
  };
}

export function formatCurrency(amount: number): string {
  return new Intl.NumberFormat("en-US", {
    style: "currency",
    currency: "USD",
    minimumFractionDigits: 0,
    maximumFractionDigits: 0,
  }).format(amount);
}
```

import { loadSchools, filterSchoolsByState, getTopSchools } from "./loadSchools";
import { loadEarnings, findEarningsForSchool } from "./loadEarnings";

import type { School } from "./schema";

export interface SchoolWithEarnings extends School {
  startingSalary: number | null;
  medianEarnings10yr: number | null;
}

export function getSchoolsWithEarnings(): SchoolWithEarnings[] {
  const schools = loadSchools();
  const earnings = loadEarnings();

  return schools.map((school) => {
    const e = earnings.find((e) => e.schoolName === school.name);
    return {
      ...school,
      startingSalary: e?.startingSalary ?? null,
      medianEarnings10yr: e?.medianEarnings10yr ?? null,
    };
  });
}

export function getTop100WithEarnings(): SchoolWithEarnings[] {
  const all = getSchoolsWithEarnings();
  return [...all].sort((a, b) => a.ranking - b.ranking).slice(0, 100);
}

export function getSchoolsByStateWithEarnings(state: string): SchoolWithEarnings[] {
  const all = getSchoolsWithEarnings();
  return all.filter((s) => s.state.toLowerCase() === state.toLowerCase());
}
